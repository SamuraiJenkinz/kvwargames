Create the module. Structure:

```python
"""
Text preprocessor for TTS input.

Pipeline order (FIXED — re-ordering breaks test cases):
  1. markdown_strip — remove emphasis markers, headers, bullets, backticks, HTML tags
  2. acronym_expand — letter-space wargame acronyms via explicit dict (word-boundary, case-sensitive)
  3. number_normalize — years/ordinals/percentages/plain integers via num2words

Why this order:
  - markdown must strip FIRST so `**EDIP**` doesn't become `**E D I P**`
  - acronyms expand BEFORE numbers so `PC 2024` becomes `P C 2024` (acronym expanded, year still intact for year-detection)

Unknown acronyms pass through unchanged (the golden-file corpus is the safety net;
new acronyms surface during Phase-16 listen-through and are added in a follow-up).

See `.planning/phases/13-firewall-spike-mockable-backend-foundation/13-CONTEXT.md`
§decisions.preprocessor for the locked design.
"""

import re
from num2words import num2words


# ---------------------------------------------------------------------------
# Acronym dictionary — explicit entries only, no heuristic fallback.
# Pluralized forms are explicit entries (no runtime suffix logic).
# Word-boundary match, case-sensitive.
# ---------------------------------------------------------------------------
ACRONYMS: dict[str, str] = {
    "EDIPs": "E D I Ps",
    "EDIP": "E D I P",
    "PCs": "P Cs",
    "PC": "P C",
    "POs": "P Os",
    "PO": "P O",
    "CRM": "C R M",
    "ICs": "I Cs",
    "IC": "I C",
    "LEFS": "L E F S",
    "SIEP": "S I E P",
    "SoS": "S O S",  # pronounced letter-by-letter (emergency/military convention)
    "EU": "E U",
    "NATO": "Nato",  # pronounced as word (Title-case hints TTS)
}
# IMPORTANT: longer forms (EDIPs) before shorter (EDIP) — for regex alternation precedence.
# Python dict preserves insertion order since 3.7; the iteration order above is authoritative.


def _strip_markdown(text: str) -> str:
    """Remove markdown and HTML emphasis while preserving sentence terminators."""
    # HTML tags first (they might contain chars that look like markdown)
    text = re.sub(r"<[^>]+>", "", text)
    # Backticks
    text = text.replace("`", "")
    # Line-start headers (# Heading, ## Subhead, ...)
    text = re.sub(r"(?m)^#+\s+", "", text)
    # Line-start bullets (- item, * item, 1. item, 2. item, ...)
    text = re.sub(r"(?m)^(?:[-*]\s+|\d+\.\s+)", "", text)
    # Bold/italic markers — strip the characters, keep enclosed text
    text = text.replace("**", "")
    # Single * and _ (italic) — only when surrounded by non-space context to avoid
    # eating bullet asterisks we already handled, but simplest safe approach:
    text = re.sub(r"\*(?=\S)|(?<=\S)\*", "", text)
    text = re.sub(r"_(?=\S)|(?<=\S)_", "", text)
    return text


def _expand_acronyms(text: str) -> str:
    """Replace wargame acronyms with their letter-spaced or phonetic expansions."""
    # Build an alternation pattern in insertion order (longest first for EDIPs-vs-EDIP).
    pattern = r"\b(" + "|".join(re.escape(k) for k in ACRONYMS.keys()) + r")\b"

    def replace(match: re.Match) -> str:
        return ACRONYMS[match.group(1)]

    return re.sub(pattern, replace, text)


def _normalize_numbers(text: str) -> str:
    """Convert years, ordinals, percentages, and plain integers to spelled-out words."""
    # Order: year → ordinal → percentage → plain (non-overlapping by regex guards)
    # 1. Years 1900-2099
    def year_sub(m: re.Match) -> str:
        return num2words(int(m.group(0)), to="year")

    text = re.sub(r"\b(?:19|20)\d{2}\b", year_sub, text)

    # 2. Ordinals: capture number prefix, pass to num2words(n, to='ordinal')
    def ord_sub(m: re.Match) -> str:
        return num2words(int(m.group(1)), to="ordinal")

    text = re.sub(r"\b(\d+)(?:st|nd|rd|th)\b", ord_sub, text)

    # 3. Percentages: capture number (int or decimal), emit "<num> percent"
    def pct_sub(m: re.Match) -> str:
        raw = m.group(1)
        n = float(raw) if "." in raw else int(raw)
        return f"{num2words(n)} percent"

    text = re.sub(r"\b(\d+(?:\.\d+)?)%", pct_sub, text)

    # 4. Plain integers / decimals (catch-all — runs LAST so prior rules own their regions)
    def num_sub(m: re.Match) -> str:
        raw = m.group(0)
        n = float(raw) if "." in raw else int(raw)
        return num2words(n)

    text = re.sub(r"\b\d+(?:\.\d+)?\b", num_sub, text)

    return text


def preprocess(text: str) -> str:
    """Apply the fixed preprocessing pipeline: strip markdown, expand acronyms, normalize numbers."""
    text = _strip_markdown(text)
    text = _expand_acronyms(text)
    text = _normalize_numbers(text)
    return text
```

Implementation notes that MUST be respected in the final file:
- `ACRONYMS` is defined in insertion order with longer forms FIRST (EDIPs before EDIP, etc.), so the alternation regex matches the longer form greedily. Python 3.7+ dicts preserve insertion order.
- `_normalize_numbers` runs year → ordinal → percentage → plain-int in that order. Each prior substitution REMOVES matched digits from the string (replaces with spelled-out form), so later regexes don't re-match them. The catch-all at the end only sees integers not already consumed.
- The single-`*` and single-`_` strippers use `(?=\S)|(?<=\S)` lookaround to avoid eating asterisks at bullet line starts (which `_strip_markdown`'s line-start-bullet rule already handled).
- Do NOT use `markdown-it-py` — the regex battery above is simpler and has no tree-walking overhead. CONTEXT.md leaves this at Claude's discretion; the regex approach is chosen for zero new heavy dependencies.

def test_markdown_must_strip_before_acronym():
    # If acronym ran first, '**EDIP**' would become '**E D I P**' (stars still wrapped).
    # Correct order strips markdown first, then expands.
    assert preprocess("**EDIP**") == "E D I P"

def test_acronym_must_expand_before_number():
    # 'PC 2024' -- acronym expands to 'P C 2024', then year 2024 becomes words.
    # If number ran first, 2024 would become words and 'PC' might still be PC, but
    # the acronym regex would still match. The invariant we're testing is the
    # composed-correct output.
    result = preprocess("PC 2024")
    # Year 2024 -> "twenty twenty-four"; PC -> "P C"
    assert "P C" in result
    assert "twenty twenty-four" in result
    assert "2024" not in result
    assert "PC" not in result

def test_pluralized_acronym_preferred_over_singular():
    # EDIPs (longest-match-first) must expand to "E D I Ps", not "E D I Ps" or "EDIPs" stuck.
    assert preprocess("Multiple EDIPs deployed.") == "Multiple E D I Ps deployed."
