    ```
    GameStore {
      // App phase
      phase: AppPhase                              // initial: 'setup'
      setupMode: SetupMode                         // initial: 'home'
      setPhase: (phase: AppPhase) => void
      setSetupMode: (mode: SetupMode) => void

      // Configuration
      gameConfig: GameConfig | null                 // initial: null
      configJson: string                            // initial: JSON.stringify(EDIP_CONFIG, null, 2)
      briefText: string                             // initial: ''
      setGameConfig: (cfg: GameConfig) => void
      setConfigJson: (json: string) => void
      setBriefText: (text: string) => void

      // Game state
      gameState: GameState | null                   // initial: null
      setGameState: (state: GameState) => void
      applyStateUpdate: (update: StateUpdate) => void

      // Chat
      messages: ChatMessage[]                       // initial: []
      addMessage: (msg: ChatMessage) => void
      addMessages: (msgs: ChatMessage[]) => void
      clearMessages: () => void

      // LLM conversation history
      llmHistory: Array<{ role: 'user' | 'assistant'; content: string }>  // initial: []
      appendHistory: (role: 'user' | 'assistant', content: string) => void

      // UI state
      loading: boolean                              // initial: false
      setLoading: (loading: boolean) => void
      activeTab: 'cards' | 'actions' | 'guide'     // initial: 'cards'
      setActiveTab: (tab: 'cards' | 'actions' | 'guide') => void

      // Actions
      initGame: (config: GameConfig, scenarioIndex: number) => void
      resetGame: () => void
    }
    ```

    ACTION IMPLEMENTATIONS:

    1. **Simple setters** (setPhase, setSetupMode, setGameConfig, setConfigJson, setBriefText, setGameState, setLoading, setActiveTab): Direct assignment via immer `set`.

    2. **addMessage**: Push to `state.messages` array.

    3. **addMessages**: Push all to `state.messages` array via spread or concat.

    4. **clearMessages**: Set `state.messages = []`.

    5. **appendHistory**: Push `{ role, content }` to `state.llmHistory`.

    6. **initGame(config, scenarioIndex)**: (from research Pattern 6)
       ```
       set((state) => {
         const scenario = config.scenarios[scenarioIndex]
         state.gameConfig = config
         state.gameState = {
           round: 1,
           scenarioIndex,
           crisisSeverity: scenario.startState.crisisSeverity,
           crisisState: scenario.startState.crisisState,
           edipLegitimacy: scenario.startState.edipLegitimacy,
           teams: config.teams.map((t) => ({
             id: t.id, name: t.name,
             pc: t.pc, po: t.po, readiness: t.readiness,
             stock: t.stock, crm: t.crm, ic: t.ic,
           })),
           cardsThisRound: [],
         }
         state.phase = 'game'
         state.messages = []
         state.llmHistory = []
       })
       ```

         state.briefText = ''
         state.gameState = null
         state.messages = []
         state.llmHistory = []

       - Guard: if `!state.gameState` return (no-op)

         - stock: clamp [0, 99]
         - crm: clamp [0, 99]
         - ic: clamp [0, 99]
       - Use `!= null` (not `!== null`) to catch both null and undefined

       ```typescript
       import { act } from '@testing-library/react'
       import { afterEach } from 'vitest'
       import { create as actualCreate } from 'zustand'
       import type { StateCreator } from 'zustand'

       const storeResetFns = new Set<() => void>()

       const create = (<T,>(stateCreator: StateCreator<T>) => {
         const store = actualCreate(stateCreator)
         const initialState = store.getState()
         storeResetFns.add(() => store.setState(initialState, true))
         return store
       }) as typeof actualCreate

       afterEach(() => {
         act(() => {
           storeResetFns.forEach((resetFn) => resetFn())
         })
       })

       export { create }
       export default create
       ```
