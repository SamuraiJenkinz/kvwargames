    ```python
    """
    Backend error-injection tests for Phase 08-04.

    Scenarios (using httpx.MockTransport — same pattern as test_llm_auth_header.py):
    1. Upstream returns 504 with LLM_TIMEOUT error code  -> backend returns 504
    2. Upstream returns 500 with LLM_UPSTREAM_ERROR code -> backend returns 500
    3. Upstream returns 200 with truncated/malformed body -> backend returns 500 INTERNAL_ERROR (KeyError on data["choices"][0]["message"]["content"])
    """
    import httpx
    import pytest
    from fastapi.testclient import TestClient

    from app.main import app


    def _build_mock_client_error(status: int, code: str, message: str) -> httpx.AsyncClient:
        def handler(request: httpx.Request) -> httpx.Response:
            return httpx.Response(status, json={"error": {"code": code, "message": message}})
        return httpx.AsyncClient(transport=httpx.MockTransport(handler))


    def _build_mock_client_truncated() -> httpx.AsyncClient:
        def handler(request: httpx.Request) -> httpx.Response:
            # 200 OK but body cuts off mid-field — choices[0].message.content is missing
            return httpx.Response(200, content=b'{"choices": [{"message": {"role": "assistant", "content"')
        return httpx.AsyncClient(transport=httpx.MockTransport(handler))


    def _post_trivial_llm_request(client: TestClient) -> httpx.Response:
        # Minimal request body to exercise the /api/llm route. Adjust field names
        # if the actual schema differs — read backend/app/routers/llm.py for the
        # current request model. Pattern mirrors test_llm_auth_header.py.
        return client.post(
            "/api/llm",
            json={
                "messages": [{"role": "user", "content": "ping"}],
            },
        )


    def test_upstream_timeout_returns_504(env_base):
        with TestClient(app) as client:
            original = app.state.http_client
            app.state.http_client = _build_mock_client_error(504, "LLM_TIMEOUT", "upstream timed out")
            try:
                resp = _post_trivial_llm_request(client)
            finally:
                app.state.http_client = original
        assert resp.status_code == 504
        body = resp.json()
        # Backend should propagate the upstream error code; exact envelope shape
        # depends on the router's error handler. Adjust assertion if it differs.
        assert "LLM_TIMEOUT" in str(body) or body.get("error", {}).get("code") == "LLM_TIMEOUT"


    def test_upstream_500_returns_500(env_base):
        with TestClient(app) as client:
            original = app.state.http_client
            app.state.http_client = _build_mock_client_error(500, "LLM_UPSTREAM_ERROR", "upstream failed")
            try:
                resp = _post_trivial_llm_request(client)
            finally:
                app.state.http_client = original
        assert resp.status_code == 500
        body = resp.json()
        assert "LLM_UPSTREAM_ERROR" in str(body) or body.get("error", {}).get("code") == "LLM_UPSTREAM_ERROR"


    def test_truncated_body_returns_500_internal_error(env_base):
        with TestClient(app) as client:
            original = app.state.http_client
            app.state.http_client = _build_mock_client_truncated()
            try:
                resp = _post_trivial_llm_request(client)
            finally:
                app.state.http_client = original
        assert resp.status_code == 500
        body = resp.json()
        # Catch-all path — exact code may be INTERNAL_ERROR or generic; assert the
        # backend did NOT crash with an unhandled exception (status would be raw 500
        # with no JSON body in that case).
        assert isinstance(body, dict)
    ```

    ```python
    """
    Backend startup-failure test for Phase 08-04.

    When LLM_API_KEY is not set, get_settings() must raise pydantic ValidationError
    at startup (NOT a runtime 500). Per 08-RESEARCH Pitfall 3, this is asserted at
    the settings layer directly — we do NOT enter TestClient(app), because the
    lifespan calls get_settings() and crashes before the test body runs.
    """
    import pytest
    from pydantic import ValidationError

    from app.config import get_settings


    def test_missing_llm_api_key_raises_validation_error(monkeypatch):
        # env_base fixture (autouse) sets LLM_API_KEY; explicitly remove it.
        monkeypatch.delenv("LLM_API_KEY", raising=False)
        # Clear settings cache so the next get_settings() call re-reads env.
        # If get_settings is @lru_cache-wrapped (per app/config.py), this is the
        # standard cache_clear() entry point. If conftest's reset_settings_cache
        # autouse already handled this, the explicit call is a defensive no-op.
        if hasattr(get_settings, "cache_clear"):
            get_settings.cache_clear()

        with pytest.raises(ValidationError):
            get_settings()
    ```

    **Adaptation note**: if `get_settings` is NOT `@lru_cache`-wrapped (e.g. it's a plain function or uses a module-level cache pattern), the `cache_clear()` call must change accordingly. Read `app/config.py` first; the test must use whatever the actual cache-clear mechanism is.
