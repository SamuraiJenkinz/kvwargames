**File 1: `src/components/game/StatePanel/TrackBar.tsx`**

Two modes — simple (left-growing) and centre-zero. Props:

```typescript
interface TrackBarProps {
  label: string           // e.g. "Severity"
  value: number
  min: number
  max: number
  colorClass: string      // e.g. "bg-track-severity" or "bg-track-legitimacy"
  mode?: 'simple' | 'center-zero'  // default 'simple'
}
```

Layout:
- Outer: `<div className="space-y-1">`
- Header row: `<div className="flex justify-between text-xs font-mono"><span className="text-text-muted uppercase">{label}</span><span>{formattedValue}</span></div>`
  - For legitimacy (signed), formattedValue is `value >= 0 ? \`+${value}\` : \`${value}\``. For severity, just `${value}`.
- Bar shell: `<div className="w-full h-1.5 bg-bg-surface rounded-sm overflow-hidden relative">`

Simple mode:
- Single fill `<div className={cn('h-full rounded-sm transition-[width] duration-300 ease-out', colorClass)} style={{ width: \`${((value - min) / (max - min)) * 100}%\` }} />`

Centre-zero mode (legitimacy −2 to +2):
- Centre line `<div className="absolute left-1/2 top-0 bottom-0 w-px bg-border-default" />`
- Fill positioned from centre:
  ```tsx
  const magnitude = Math.abs(value)
  const halfSpan = (max - min) / 2  // 2 for legitimacy
  const widthPct = (magnitude / halfSpan) * 50  // % of full bar
  const leftPct = value >= 0 ? 50 : 50 - widthPct
  <div
    className={cn('absolute top-0 bottom-0 rounded-sm transition-all duration-300 ease-out', colorClass)}
    style={{ left: `${leftPct}%`, width: `${widthPct}%` }}
  />
  ```

Per research Open Question 3, centre-zero is recommended for legitimacy to communicate signed value.

**File 2: `src/components/game/StatePanel/PcBadge.tsx`**

```tsx
import { getPcBadge } from '@/lib/pcThresholds'

export default function PcBadge({ pc }: { pc: number }) {
  const state = getPcBadge(pc)
  if (state === null) return null

  if (state === 'CRISIS') {
    return (
      <span
        className="text-[10px] font-mono px-1.5 py-0.5 rounded-sm bg-crisis-security/20 text-crisis-security"
        style={{ animation: 'var(--animate-blink)' }}
      >
        CRISIS
      </span>
    )
  }
  return (
    <span className="text-[10px] font-mono px-1.5 py-0.5 rounded-sm bg-persona-finch/20 text-persona-finch">
      STRAINED
    </span>
  )
}
```

**File 3: `src/components/game/StatePanel/PersonaDots.tsx`**

```tsx
import { useGameStore } from '@/lib/gameStore'
import { PERSONA_ORDER, PERSONA_META } from '@/lib/personaConfig'
import { getPersonasThisRound } from '@/lib/pcThresholds'

export default function PersonaDots() {
  const messages = useGameStore(s => s.messages)
  const spoken = getPersonasThisRound(messages)

  return (
    <div className="flex gap-2 items-center">
      {PERSONA_ORDER.map((id) => {
        const meta = PERSONA_META[id]
        const lit = spoken.has(id)
        return (
          <div key={id} className="flex items-center gap-1">
            <div
              className={cn(
                'w-2 h-2 rounded-full transition-opacity duration-300',
                meta.dotClass,
                lit ? 'opacity-100' : 'opacity-25'
              )}
              data-testid={`persona-dot-${id}`}
              data-lit={lit}
            />
            <span className={cn('text-[10px] font-mono uppercase', lit ? 'text-text-primary' : 'text-text-muted')}>
              {meta.displayName}
            </span>
          </div>
        )
      })}
    </div>
  )
}
```

```tsx
import type { TeamState } from '@/types/game'
import PcBadge from './PcBadge'

const FIELDS: Array<{ key: keyof TeamState; label: string; colorClass: string }> = [
  { key: 'pc',        label: 'PC',  colorClass: 'text-resource-pc' },
  { key: 'po',        label: 'PO',  colorClass: 'text-resource-po' },
  { key: 'readiness', label: 'RDY', colorClass: 'text-resource-readiness' },
  { key: 'stock',     label: 'STK', colorClass: 'text-resource-stock' },
  { key: 'crm',       label: 'CRM', colorClass: 'text-resource-crm' },
  { key: 'ic',        label: 'IC',  colorClass: 'text-resource-ic' },
]

export default function TeamCard({ team }: { team: TeamState }) {
  // Short name: strip "Team X: " prefix if present
  const short = `Team ${team.id}`
  return (
    <div className="bg-bg-surface border border-border-subtle rounded-sm p-2">
      <div className="flex items-center justify-between mb-2">
        <span className="font-mono text-xs uppercase tracking-wide">{short}</span>
        <PcBadge pc={team.pc} />
      </div>
      <div className="grid grid-cols-2 gap-x-2 gap-y-1">
        {FIELDS.map(f => {
          const val = team[f.key] as number
          const formatted = f.key === 'po' ? (val >= 0 ? `+${val}` : `${val}`) : `${val}`
          return (
            <div key={f.key} className="flex items-baseline justify-between">
              <span className="text-[10px] text-text-muted font-mono uppercase">{f.label}</span>
              <span className={cn('text-sm font-medium font-mono', f.colorClass)}>{formatted}</span>
            </div>
          )
        })}
      </div>
    </div>
  )
}
```

**File 2: `src/components/game/StatePanel/StatePanel.tsx`** — REPLACE the stub.

```tsx
import { useGameStore } from '@/lib/gameStore'
import TrackBar from './TrackBar'
import TeamCard from './TeamCard'
import PersonaDots from './PersonaDots'

export default function StatePanel() {
  const gameState = useGameStore(s => s.gameState)
  if (!gameState) return null

  return (
    <div
      data-testid="state-panel"
      className="w-[210px] flex-none overflow-y-auto border-r border-border-subtle p-3 space-y-4"
    >
      <TrackBar
        label="Severity"
        value={gameState.crisisSeverity}
        min={0}
        max={5}
        colorClass="bg-track-severity"
        mode="simple"
      />
      <TrackBar
        label="Legitimacy"
        value={gameState.edipLegitimacy}
        min={-2}
        max={2}
        colorClass="bg-track-legitimacy"
        mode="center-zero"
      />

      <div className="pt-2 border-t border-border-subtle">
        <div className="text-[10px] text-text-muted font-mono uppercase tracking-wide mb-2">Personas this round</div>
        <PersonaDots />
      </div>

      <div className="pt-2 border-t border-border-subtle space-y-2">
        {gameState.teams.map(team => (
          <TeamCard key={team.id} team={team} />
        ))}
      </div>
    </div>
  )
}
```
